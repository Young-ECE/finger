#提升串口权限#

    sudo chmod 666 /dev/ttyACM0 

    #永久提升权限,创建规则文件
    sudo nano /etc/udev/rules.d/99-ttyacm.rules
    KERNEL=="ttyACM[0-9]*", MODE="0666", GROUP="dialout"
    sudo udevadm control --reload-rules
    sudo udevadm trigger
